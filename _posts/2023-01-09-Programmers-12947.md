### Answer Code1(23.01.09)

``` java
import java.util.*;

class Solution {
    public boolean solution(int x) {
        boolean answer = true;
        int sum = 0;
        int number = x;

        while(number > 0) {
            int div = number % 10;
            sum += div;
            number /= 10;
        }
        if(x % sum == 0) {
            answer = true;
        } else {
            answer = false;
        }

        return answer;
    }
}
```

### Answer Code2(23.01.16)

class Solution {
    public boolean solution(int num) {
        String[] temp = String.valueOf(num).split("");

        int sum = 0;
        for (String s : temp) {
            sum += Integer.parseInt(s);
        }

        if (num % sum == 0) {
                return true;
        } else {
          return false;
        }
    }
}
